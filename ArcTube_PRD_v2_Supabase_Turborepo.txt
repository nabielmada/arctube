1. Executive Summary
Mission
ArcTube adalah platform video Pay-Per-Second (PpS) di mana pengguna membayar $0.001 USDC per detik konten yang ditonton, langsung ke kreator, real-time, tanpa subscription. Model ini hanya bisa eksis karena Arc gas ~$0.000001 — 300.000× lebih murah dari Stripe ($0.30 minimum). Dibangun dengan Supabase (database + auth + realtime) dan Turborepo (monorepo frontend + backend).

3. Supabase Database Schema
Supabase vs Prisma
Tidak ada Prisma migration. Schema didefinisikan sebagai SQL di packages/supabase/src/schema.sql dan di-run sekali via Supabase Dashboard atau CLI. TypeScript types di-generate otomatis via `supabase gen types typescript`.
	

3.1 SQL Schema (packages/supabase/src/schema.sql)
-- Enable UUID extension
create extension if not exists "uuid-ossp";


-- ─── Users (extends Supabase auth.users) ─────────────────────────────
create table public.profiles (
  id              uuid references auth.users on delete cascade primary key,
  wallet_id       text unique not null,   -- Circle Programmable Wallet ID
  wallet_address  text unique not null,   -- Arc address 0x...
  usdc_balance    numeric(18,8) default 0,
  role            text default 'viewer' check (role in ('viewer','creator','admin')),
  username        text unique,
  created_at      timestamptz default now()
);


-- Auto-create profile on auth.users insert
create or replace function public.handle_new_user()
returns trigger language plpgsql security definer as $$
begin
  insert into public.profiles (id) values (new.id);
  return new;
end;
$$;


create trigger on_auth_user_created
  after insert on auth.users
  for each row execute procedure public.handle_new_user();


-- ─── Videos ───────────────────────────────────────────────────────────
create table public.videos (
  id               uuid default uuid_generate_v4() primary key,
  creator_id       uuid references public.profiles(id) on delete cascade,
  title            text not null,
  description      text,
  duration_seconds int  not null,
  price_per_second numeric(18,8) default 0.001,   -- USDC per second
  ownership_price  numeric(18,8) not null,        -- cap for permanent access
  hls_manifest_url text not null,
  teaser_seconds   int  default 10,
  total_earned     numeric(18,8) default 0,
  status           text default 'active' check (status in ('active','draft','removed')),
  created_at       timestamptz default now()
);


-- ─── Transactions ─────────────────────────────────────────────────────
create table public.transactions (
  id               uuid default uuid_generate_v4() primary key,
  user_id          uuid references public.profiles(id),
  video_id         uuid references public.videos(id),
  amount_usdc      numeric(18,8) not null,
  arc_tx_hash      text unique,
  arc_block_number bigint,
  second_index     int not null,            -- which second this covers
  status           text default 'confirmed' check (status in ('pending','confirmed','failed')),
  created_at       timestamptz default now()
);


-- ─── Ownership ────────────────────────────────────────────────────────
create table public.ownerships (
  id               uuid default uuid_generate_v4() primary key,
  user_id          uuid references public.profiles(id),
  video_id         uuid references public.videos(id),
  total_paid       numeric(18,8) default 0,
  is_complete      boolean default false,
  completed_at     timestamptz,
  arc_mint_tx_hash text,
  unique(user_id, video_id)
);


-- ─── Watch Sessions ───────────────────────────────────────────────────
create table public.watch_sessions (
  id               uuid default uuid_generate_v4() primary key,
  user_id          uuid references public.profiles(id),
  video_id         uuid references public.videos(id),
  started_at       timestamptz default now(),
  last_heartbeat   timestamptz default now(),
  seconds_watched  int default 0,
  total_paid_usdc  numeric(18,8) default 0,
  is_active        boolean default true
);


-- ─── Indexes ──────────────────────────────────────────────────────────
create index on public.transactions(user_id, video_id);
create index on public.transactions(arc_tx_hash);
create index on public.ownerships(user_id, video_id);
create index on public.watch_sessions(user_id, is_active);


-- ─── Row Level Security ──────────────────────────────────────────────
alter table public.profiles    enable row level security;
alter table public.videos      enable row level security;
alter table public.transactions enable row level security;
alter table public.ownerships  enable row level security;
alter table public.watch_sessions enable row level security;


-- Profiles: users can read all, update only own
create policy "Public profiles"  on public.profiles for select using (true);
create policy "Own profile edit" on public.profiles for update using (auth.uid() = id);


-- Videos: public read, creator write
create policy "Public videos"    on public.videos for select using (true);
create policy "Creator insert"   on public.videos for insert with check (auth.uid() = creator_id);


-- Transactions: users see their own only
create policy "Own transactions" on public.transactions for select using (auth.uid() = user_id);
create policy "Service inserts"  on public.transactions for insert with check (true);

4. Supabase Realtime — Replacing WebSocket Server
Key Architecture Decision
Daripada membangun WebSocket server terpisah (Socket.io + Redis), kita gunakan Supabase Realtime. Setiap INSERT ke tabel `transactions` otomatis di-broadcast ke semua subscriber. Frontend subscribe ke channel `transactions:user_id=eq.{userId}` dan USDC Meter update secara real-time. Ini menghilangkan seluruh apps/api/src/payment/gateway.ts.
	

4.1 Frontend Realtime Subscription (composables/useSupabaseRealtime.ts)
// composables/useSupabaseRealtime.ts
import { supabase } from '@arctube/supabase';


export const useSupabaseRealtime = (userId: string, videoId: string) => {
  const latestTx   = ref<any>(null);
  const txCount    = ref(0);
  const totalPaid  = ref(0);


  onMounted(() => {
    const channel = supabase
      .channel(`tx-${userId}-${videoId}`)
      .on('postgres_changes', {
        event:  'INSERT',
        schema: 'public',
        table:  'transactions',
        filter: `user_id=eq.${userId}`,
      }, (payload) => {
        latestTx.value  = payload.new;
        txCount.value  += 1;
        totalPaid.value = parseFloat(payload.new.amount_usdc) * txCount.value;
      })
      .subscribe();


    onUnmounted(() => supabase.removeChannel(channel));
  });


  return { latestTx, txCount, totalPaid };
};


4.2 Ownership Realtime (auto-unlock when cap reached)
// composables/useOwnershipRealtime.ts
export const useOwnershipRealtime = (userId: string, videoId: string) => {
  const isOwned = ref(false);


  onMounted(() => {
    supabase
      .channel(`ownership-${userId}-${videoId}`)
      .on('postgres_changes', {
        event:  'UPDATE',
        schema: 'public',
        table:  'ownerships',
        filter: `user_id=eq.${userId}`,
      }, (payload) => {
        if (payload.new.is_complete === true) {
          isOwned.value = true;
          // Trigger ownership celebration UI
        }
      })
      .subscribe();
  });


  return { isOwned };
};

5. NestJS API — Payment Engine
5.1 Auth Guard (Supabase JWT)
Supabase Auth Integration
NestJS tidak perlu auth module sendiri. Setiap request membawa Supabase JWT di header Authorization. Guard memvalidasi token via Supabase Admin SDK dan inject user ke request context.
	

// src/auth/supabase-auth.guard.ts
import { Injectable, CanActivate, ExecutionContext, UnauthorizedException } from '@nestjs/common';
import { supabaseAdmin } from '@arctube/supabase';


@Injectable()
export class SupabaseAuthGuard implements CanActivate {
  async canActivate(ctx: ExecutionContext): Promise<boolean> {
    const req   = ctx.switchToHttp().getRequest();
    const token = req.headers.authorization?.replace('Bearer ', '');
    if (!token) throw new UnauthorizedException();


    const { data: { user }, error } = await supabaseAdmin.auth.getUser(token);
    if (error || !user) throw new UnauthorizedException();


    req.user = user;  // { id, email, ... }
    return true;
  }
}


5.2 Payment Service (src/payment/payment.service.ts)
import { Injectable, BadRequestException } from '@nestjs/common';
import { supabaseAdmin } from '@arctube/supabase';
import { CircleWalletService } from '../wallet/circle-wallet.service';


@Injectable()
export class PaymentService {
  constructor(private circle: CircleWalletService) {}


  async processHeartbeat(userId: string, videoId: string, second: number) {
    // ── 1. Fetch video + ownership in parallel ──────────────────────
    const [{ data: video }, { data: ownership }] = await Promise.all([
      supabaseAdmin.from('videos').select('*').eq('id', videoId).single(),
      supabaseAdmin.from('ownerships')
        .select('*').eq('user_id', userId).eq('video_id', videoId).maybeSingle()
    ]);


    if (!video) throw new BadRequestException('Video not found');
    if (ownership?.is_complete) return { success: true, isOwned: true };


    // ── 2. Fetch user profile (wallet info + balance) ───────────────
    const { data: profile } = await supabaseAdmin
      .from('profiles').select('*').eq('id', userId).single();


    if (!profile) throw new BadRequestException('Profile not found');


    const amount = parseFloat(video.price_per_second);
    if (parseFloat(profile.usdc_balance) < amount)
      throw new BadRequestException('Insufficient USDC balance');


    // ── 3. Execute Circle transfer on Arc ───────────────────────────
    const { data: creator } = await supabaseAdmin
      .from('profiles').select('wallet_address').eq('id', video.creator_id).single();


    const transfer = await this.circle.transferUSDC({
      fromWalletId: profile.wallet_id,
      toAddress:    creator!.wallet_address,
      amount:       amount.toString(),
    });


    // ── 4. Insert transaction row (triggers Supabase Realtime) ──────
    await supabaseAdmin.from('transactions').insert({
      user_id:          userId,
      video_id:         videoId,
      amount_usdc:      amount,
      arc_tx_hash:      transfer.txHash,
      arc_block_number: transfer.blockNumber,
      second_index:     second,
      status:           'confirmed',
    });


    // ── 5. Upsert ownership accumulator ─────────────────────────────
    const newTotal = (parseFloat(ownership?.total_paid || '0')) + amount;
    const isOwned  = newTotal >= parseFloat(video.ownership_price);


    await supabaseAdmin.from('ownerships').upsert({
      user_id:    userId,
      video_id:   videoId,
      total_paid: newTotal,
      is_complete: isOwned,
      ...(isOwned ? { completed_at: new Date().toISOString() } : {})
    }, { onConflict: 'user_id,video_id' });


    // ── 6. Deduct from user balance ──────────────────────────────────
    await supabaseAdmin.from('profiles').update({
      usdc_balance: parseFloat(profile.usdc_balance) - amount
    }).eq('id', userId);


    // ── 7. Increment video total_earned ─────────────────────────────
    await supabaseAdmin.from('videos').update({
      total_earned: parseFloat(video.total_earned) + amount
    }).eq('id', videoId);


    return {
      success: true,
      txHash:  transfer.txHash,
      arcBlockNumber: transfer.blockNumber,
      amountPaid:     amount,
      totalPaidVideo: newTotal,
      ownershipPrice: parseFloat(video.ownership_price),
      isOwned,
      remainingBalance: parseFloat(profile.usdc_balance) - amount,
    };
  }
}

6.1 usePaymentStream Composable
// apps/web/composables/usePaymentStream.ts
import { useSupabaseRealtime } from './useSupabaseRealtime';
import { useOwnershipRealtime } from './useOwnershipRealtime';


export const usePaymentStream = (videoId: string, userId: string) => {
  const sessionId   = ref<string|null>(null);
  const isPlaying   = ref(false);
  const currentSecond = ref(0);
  const isPaused    = ref(false);
  let heartbeatTimer: ReturnType<typeof setInterval>|null = null;


  // Subscribe to Supabase Realtime (replaces WebSocket)
  const { latestTx, txCount, totalPaid } = useSupabaseRealtime(userId, videoId);
  const { isOwned }                       = useOwnershipRealtime(userId, videoId);


  const startSession = async () => {
    const { data } = await $fetch('/api/payment/session/start', {
      method: 'POST', body: { videoId }
    });
    sessionId.value = data.sessionId;
    if (!data.isAlreadyOwned) startHeartbeat();
  };


  const startHeartbeat = () => {
    heartbeatTimer = setInterval(async () => {
      if (!isPlaying.value || isOwned.value || isPaused.value) return;


      try {
        const result = await $fetch('/api/payment/heartbeat', {
          method: 'POST',
          body: { sessionId: sessionId.value, videoId, second: currentSecond.value }
        });
        // Realtime handles UI updates via Supabase subscription above
        if (result.isOwned) clearInterval(heartbeatTimer!);
      } catch (err: any) {
        if (err.statusCode === 400) {  // insufficient balance
          isPlaying.value = false;
          isPaused.value  = true;
          clearInterval(heartbeatTimer!);
        }
      }
    }, 1000);  // every 1 second
  };


  const onTimeUpdate = (t: number) => { currentSecond.value = Math.floor(t); };
  const onPlay       = () => { isPlaying.value = true; isPaused.value = false; };
  const onPause      = () => { isPlaying.value = false; };


  onUnmounted(() => { if (heartbeatTimer) clearInterval(heartbeatTimer); });


  return { startSession, isPlaying, isOwned, totalPaid, txCount, latestTx, onTimeUpdate, onPlay, onPause };
};


6.2 Supabase Auth in Nuxt
// apps/web/plugins/supabase.client.ts
import { supabase } from '@arctube/supabase';


export default defineNuxtPlugin(() => {
  const user = useState('user', () => null);


  supabase.auth.onAuthStateChange((event, session) => {
    user.value = session?.user ?? null;
    if (event === 'SIGNED_IN' && session) {
      // Provision Circle wallet if not yet done
      $fetch('/api/auth/provision-wallet', {
        method: 'POST',
        headers: { Authorization: `Bearer ${session.access_token}` }
      });
    }
  });


  return { provide: { supabase } };
});

9.2 Economic Margin Template (copy-paste for submission)
ECONOMIC PROOF — Why This Model Fails with Traditional Payment Infrastructure
ArcTube charges $0.001 USDC per second of video watched. A typical 2-minute demo generates $0.12 in revenue across 120 separate transactions.With Stripe: Revenue $0.12 minus minimum fee $0.30 equals negative $0.18 net loss. Every single view is guaranteed to lose money. Pay-per-second pricing is mathematically impossible on Web2 infrastructure.With Arc Nanopayments: Revenue $0.12 minus total gas $0.00012 (120 transactions at ~$0.000001 each) equals $0.11988 net profit. Margin: 99.9%. The model is profitable from the very first second watched.Fee ratio difference: 300,000×. This is not a marginal improvement — it is an entirely new economic primitive. Pay-per-second content monetization cannot be added to Web2 infrastructure. It can only exist on Arc.

11. Quick-Start Commands
11.1 Bootstrap (run once)
# 1. Clone + install
git clone https://github.com/yourname/arctube && cd arctube
pnpm install


# 2. Copy env
cp .env.example .env
# Fill in: SUPABASE_URL, SUPABASE_ANON_KEY, SUPABASE_SERVICE_ROLE_KEY,
#          CIRCLE_API_KEY, CIRCLE_WALLET_SET_ID, ARC_USDC_CONTRACT


# 3. Push Supabase schema
# Option A — Supabase Dashboard SQL Editor: paste packages/supabase/src/schema.sql
# Option B — Supabase CLI:
npx supabase db push


# 4. Generate TypeScript types from Supabase
npx supabase gen types typescript --project-id YOUR_PROJECT_ID \
  > packages/supabase/src/types.ts


# 5. Build shared packages
pnpm build --filter=@arctube/supabase --filter=@arctube/types --filter=@arctube/utils

pnpm dev

# └── apps/api  → http://localhost:3001 (NestJS)


# Rebuild shared packages after changes
pnpm build --filter=@arctube/supabase
